def fields(value_or_name)
  value = case value_or_name
    when Hash
      value_or_name
    when Array
      value_or_name.reduce({}) { |sum, item| sum.update item.to_sym => {}; sum }
    else
  end

  (@value[:fields] ||= {}).update value
  self
end
