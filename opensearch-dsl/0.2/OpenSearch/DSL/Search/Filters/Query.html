Overview

A filter which wraps a query so it can be used as a filter

Examples:


search do
  query do
    filtered do
      filter do
        query do
          query_string :title do
            query 'Ruby OR Python'
          end
        end
      end
    end
  end
end

def to_hash
  hash = super
  if @query
    _query = @query.respond_to?(:to_hash) ? @query.to_hash : @query
    hash[self.name].update(_query)
  end
  hash
end
