def call
  @block.arity < 1 ? self.instance_eval(&@block) : @block.call(self) if @block && ! @_block_called
  @_block_called = true
  self
end

def to_hash(options={})
  case
  when @value.empty? && ! @block
    @hash = super
  when @block
    call
    @hash = { name.to_sym => @value.map { |i| i.to_hash } }
  else
    @hash = { name.to_sym => @value }
  end
  @hash
end
