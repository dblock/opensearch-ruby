def method_missing(name, *args, &block)
  klass = Utils.__camelize(name)
  if Filters.const_defined? klass
    @value << Filters.const_get(klass).new(*args, &block)
  else
    raise NoMethodError, "undefined method '#{name}' for #{self}"
  end
end
